## Summary

Replace the current Max All implementation with a 3-step pipeline:
1. Frustum fit (existing layout)
2. Expand card DOM to fill slot (new)
3. Fit terminal to card (existing `optimizeTermToCard`)

Remove the `_fillSlot` mechanism which tried to combine steps 2+3 into a single async round-trip.

### New code:
```javascript
function maxAllFocused() {
  if (focusedSessions.size < 2) return;
  slotOrder.clear();

  // Step 1: Frustum-fit letterbox — cards positioned at Z-depth, _slotFit set
  calculateFocusedLayout();

  // Pin slot assignments before mutation
  for (var [fn, ft] of terminals) {
    if (focusedSessions.has(fn) && ft._slotIndex != null) {
      slotOrder.set(fn, ft._slotIndex);
    }
  }

  // Step 2: Expand card DOM to fill slot (mutate unconstrained dimension)
  for (var name of focusedSessions) {
    var t = terminals.get(name);
    if (!t || !t._slotFit) continue;
    var fit = t._slotFit;
    var newW = t.baseCardW;
    var newH = t.baseCardH;

    if (fit.constrainedBy === 'width') {
      // Height is letterboxed — expand card height to fill slot
      newH = Math.round(t.baseCardH * (fit.slotH / fit.fitH));
    } else {
      // Width is letterboxed — expand card width to fill slot
      newW = Math.round(t.baseCardW * (fit.slotW / fit.fitW));
    }
    t.baseCardW = newW;
    t.baseCardH = newH;
    t.dom.style.width = newW + 'px';
    t.dom.style.height = newH + 'px';
    var inner = t.dom.querySelector('.terminal-inner');
    if (inner) {
      inner.style.width = newW + 'px';
      inner.style.height = newH + 'px';
    }
  }

  // Step 3: Fit terminal to card — terminal cols/rows resize to fill new card aspect
  for (var name of focusedSessions) {
    var t = terminals.get(name);
    if (!t) continue;
    t._lockCardSize = true;  // prevent resize response from changing card size
    optimizeTermToCard(t);
  }
}
```

**Key differences:**
- No call to `maximizeCardToSlot` — replaced by inline step 2
- No second `calculateFocusedLayout()` — Z-depth from step 1 is already correct
- Pin slotOrder between step 1 and 2 (before mutation, not after)
- Step 2 is pure DOM: no terminal resize, no async, no `_fillSlot`
- Step 3 uses `optimizeTermToCard` with `_lockCardSize = true`
